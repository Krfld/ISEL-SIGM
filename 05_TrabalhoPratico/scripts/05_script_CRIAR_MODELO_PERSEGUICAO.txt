-- Suporte 'a nocao de perseguicao
CREATE TABLE pp(
	id_perseguidor int,
	id_alvo int,
	constraint fk_id_perseguidor foreign key (id_perseguidor) references cinematica(id),
	constraint fk_id_alvo foreign key (id_alvo) references cinematica(id)
);
